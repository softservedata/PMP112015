#include<iostream>
#include<algorithm>
#include"task.h"
using namespace std;

int maxSymdols(char **r,int m,int n)
{
	int group=0;
	int max=0;
	int countsymbols;
	for (int k=0;k<m;k++)
	{
		for(int i=0;i<n;i++)
		{
			countsymbols=count(r[k],r[k]+n,r[k][i]);
			if(countsymbols>max)
			{
				max=countsymbols;
				group=k;
			}
		}
	}
	return group;
}

#include<iostream>
#include"print.h"
using namespace std;

void printMatrix(char **r,int m,int n)
{
	cout<<"print groups:"<<endl;
	for(int k=0;k<m;k++)
	{
		for(int i=0;i<n;i++)
		{
			cout<<r[k][i]<<"  ";
		}
		cout<<endl;
	}
}

void printGroup(char **r,int n,int group)
{
	cout<<"Group with max number of common elements: "<<endl;
	for(int i=0;i<n;i++)
	{
		cout<<r[group][i]<<"  ";
	}
	cout<<endl;
}
