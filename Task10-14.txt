#include<iostream>
#include"Set.h"
using namespace std;

void Set:: create(int amound)
{
	n=amound;
	a=new int[n];
	k=0;
}
void Set:: addElem(int elem)
{	
	bool exist=false;
	for(int i=0;i<k;i++)
	{
		if(a[i]==elem)
		{
			exist=true;
			break;
		}
	}
	if(!exist)
	{
		a[k]=elem;
		k++;
	}
}
int Set:: getCount()
{ return k;}
void Set::  del()
{ delete[] a;}
